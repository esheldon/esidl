PRO dirsep, fname, dir, file

  ;; separate the file name from the directory in UNIX

  IF n_params() LT 1 THEN BEGIN
      print,'-Syntax: dirsep, fname, dir, file'
      return
  ENDIF

length = strlen(fname)

pos=0
pos = strpos(fname, '/', pos)
posold = pos
WHILE pos NE -1 DO BEGIN
posold = pos
pos = pos+1
pos = strpos(fname, '/', pos)
ENDWHILE

dir = strmid(fname, 0, posold+1)
file = strmid(fname, posold+1, length-posold)

return
END
